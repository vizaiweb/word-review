// ====================== 全局狀態變量 ====================== let allWords = []; let filteredWords = []; let currentWordIdx = 0; let currentFileName = ""; let currentLevel = ""; let allSentences = []; let currentSentenceIdx = 0; let currentMode = "local"; const synth = window.speechSynthesis; // 朗讀相關變量 let isWordReading = false; let isSentenceReading = false; let currentWordReadButton = null; let currentSentenceReadButton = null; let isStopping = false; // ====================== 發音品質升級核心 ====================== /** * 獲取高品質語音引擎 (優先 Google US English) */ function getHighQualityVoice() { return new Promise(resolve => { let voices = synth.getVoices(); const findVoice = () => { const availableVoices = synth.getVoices(); // 優先權順序：Google 美式英語 > Apple/自然美式英語 > 任何美式女聲 return availableVoices.find(v => v.name === 'Google US English') || availableVoices.find(v => v.name.includes('Natural') && v.lang === 'en-US') || availableVoices.find(v => v.name.includes('Samantha')) || availableVoices.find(v => v.lang.includes('en-US')) || availableVoices[0]; }; if (voices.length) resolve(findVoice()); else { synth.onvoiceschanged = () => resolve(findVoice()); } }); } /** * 核心朗讀函數：優化咬字與平滑度 */ async function speakTextWithCallback(text, onEnd) { if (!text || isStopping) { if (onEnd) onEnd(); return; } const utterance = new SpeechSynthesisUtterance(text); const voice = await getHighQualityVoice(); if (voice) utterance.voice = voice; // 設定最佳人聲參數 utterance.lang = "en-US"; utterance.rate = 0.85; // 稍微放慢一點，咬字更清晰 utterance.pitch = 1.0; // 保持標準音高，避免機器人感 utterance.volume = 1.0; utterance.onend = () => { if (!isStopping && onEnd) onEnd(); }; utterance.onerror = () => { if (!isStopping && onEnd) onEnd(); }; synth.speak(utterance); } function stopAllReading() { isStopping = true; synth.cancel(); isWordReading = false; isSentenceReading = false; if (currentWordReadButton) { currentWordReadButton.textContent = "🔊 Read 3x"; currentWordReadButton.classList.remove('reading-disabled'); } if (currentSentenceReadButton) { currentSentenceReadButton.textContent = "🔊 Read 3x"; currentSentenceReadButton.classList.remove('reading-disabled'); } setTimeout(() => { isStopping = false; }, 300); } function startReadingCycle(text, btn, isWord) { stopAllReading(); setTimeout(() => { if (isWord) { isWordReading = true; currentWordReadButton = btn; } else { isSentenceReading = true; currentSentenceReadButton = btn; } btn.textContent = "⏹️ Stop"; btn.classList.add('reading-disabled'); let count = 0; const read = () => { if (isStopping || count >= 3) { stopAllReading(); return; } speakTextWithCallback(text, () => { count++; setTimeout(read, 500); // 每次重複停頓 0.5 秒，更自然 }); }; read(); }, 350); } // ====================== GitHub 分支路徑工具 ====================== function getRawBaseUrl() { if (window.location.protocol === 'file:') return 'https://raw.githubusercontent.com/vizaiweb/word-review/main'; const host = window.location.hostname; const path = window.location.pathname; return (host.includes('dev') || path.includes('dev')) ? 'https://raw.githubusercontent.com/vizaiweb/word-review/dev' : 'https://raw.githubusercontent.com/vizaiweb/word-review/main'; } function removeFileExtension(filename) { return filename.replace(/\.(xlsx|xls|csv)$/i, ''); } function getFileListUrl(level) { return `${getRawBaseUrl()}/data/${level}/fileList.json`; } function getXlsxFileUrl(level, filename) { return `${getRawBaseUrl()}/data/${level}/${filename}`; } // ====================== 模式與邏輯 ====================== function toggleMode(mode) { currentMode = mode; stopAllReading(); allWords = []; filteredWords = []; allSentences = []; currentFileName = ""; currentLevel = ""; const fileRow = document.getElementById('fileRow'); const externalRow = document.getElementById('externalUrlRow'); const levelRow = document.getElementById('levelRow'); const toggleBtn = document.getElementById('modeToggleBtn'); const wordContent = document.getElementById("wordContent"); const sentenceArea = document.getElementById("sentenceArea"); const dayRow = document.getElementById("dayRow"); const showAllBtn = document.getElementById("showAllBtn"); if (mode === "local") { fileRow.style.display = 'flex'; externalRow.style.display = 'none'; levelRow.style.display = 'flex'; toggleBtn.textContent = "📁 Built-in DB"; toggleBtn.classList.add('active'); wordContent.innerHTML = '

✨ Select Level & File to start ✨

'; } else { fileRow.style.display = 'none'; externalRow.style.display = 'flex'; levelRow.style.display = 'none'; toggleBtn.textContent = "📂 Local File"; toggleBtn.classList.remove('active'); wordContent.innerHTML = '

✨ Select a local Excel file to start ✨

'; } sentenceArea.style.display = 'none'; dayRow.style.display = 'none'; showAllBtn.style.display = 'none'; document.getElementById("infoTipContainer").innerHTML = ''; } async function parseExcel(buf, name) { try { const wb = XLSX.read(buf, { type: "array" }); const data = XLSX.utils.sheet_to_json(wb.Sheets[wb.SheetNames[0]]); allWords = data.filter(i => i.word && i.meaning && i.day).map(i => ({ word: String(i.word).trim(), meaning: String(i.meaning).trim(), day: Number(i.day) })); allSentences = []; if (wb.SheetNames.length >= 2) { const sData = XLSX.utils.sheet_to_json(wb.Sheets[wb.SheetNames[1]]); sData.forEach(row => { let en = row.sentence || row.sentence_en || row.en || row.English; let zh = row.chinese || row.meaning || row.zh || row.Chinese; if (en) allSentences.push({ sentence_en: String(en).trim(), sentence_zh: zh ? String(zh).trim() : "Practice" }); }); } filteredWords = [...allWords]; currentWordIdx = 0; currentSentenceIdx = 0; if (allWords.length) { showWord(); document.getElementById("dayRow").style.display = 'flex'; document.getElementById("showAllBtn").style.display = 'block'; } if (allSentences.length) { document.getElementById("sentenceArea").style.display = 'block'; updateSentenceUI(); } else { document.getElementById("sentenceArea").style.display = 'none'; } updateInfoTip(); return true; } catch (e) { console.error("Parse Error", e); return false; } } // ====================== UI 渲染 ====================== function showWord() { const container = document.getElementById("wordContent"); if (!filteredWords.length) { container.innerHTML = '

No words found.

'; return; } if (currentWordIdx >= filteredWords.length) { container.innerHTML = '

🎉 Task Finished!

'; return; } const w = filteredWords[currentWordIdx]; container.innerHTML = `
💡 ${w.meaning}
${w.word.toUpperCase()}
`; document.getElementById("btnShowWord").onclick = () => document.getElementById("currentWordSpan").style.display = "block"; document.getElementById("btnReadWord").onclick = (e) => startReadingCycle(w.word, e.target, true); document.getElementById("btnPrevWord").onclick = () => { currentWordIdx--; showWord(); updateInfoTip(); }; document.getElementById("btnNextWord").onclick = () => { currentWordIdx++; showWord(); updateInfoTip(); }; } function updateSentenceUI() { const sent = allSentences[currentSentenceIdx]; const area = document.getElementById("sentenceContent"); area.innerHTML = `
📖 ${sent.sentence_zh}
${sent.sentence_en}
📌 ${currentSentenceIdx + 1} / ${allSentences.length}
`; document.getElementById("btnShowSent").onclick = () => document.getElementById("sentEn").style.display = "block"; document.getElementById("btnReadSent").onclick = (e) => startReadingCycle(sent.sentence_en, e.target, false); document.getElementById("btnPrevSent").onclick = () => { currentSentenceIdx--; updateSentenceUI(); }; document.getElementById("btnNextSent").onclick = () => { if (currentSentenceIdx < allSentences.length - 1) { currentSentenceIdx++; updateSentenceUI(); } else alert("🎉 All sentences done!"); }; document.getElementById("showAllSentencesBtn").onclick = () => { let text = allSentences.map((s, i) => `${i+1}. ${s.sentence_en}\n ${s.sentence_zh}`).join('\n\n'); alert(text); }; } function updateInfoTip() { const tip = document.getElementById('infoTipContainer'); if (!filteredWords.length) { tip.innerHTML = ''; return; } const cur = filteredWords[currentWordIdx]; tip.innerHTML = `
${removeFileExtension(currentFileName)} | Day ${cur?.day || '--'} | ${currentWordIdx + 1}/${filteredWords.length} words
`; } // ====================== 事件初始化 ====================== document.addEventListener('DOMContentLoaded', () => { const levelSelect = document.getElementById('levelSelect'); const fileSelect = document.getElementById('fileSelect'); const localFileInput = document.getElementById('localFileInput'); document.getElementById('modeToggleBtn').onclick = () => toggleMode(currentMode === "local" ? "external" : "local"); document.getElementById('levelConfirm').onclick = async () => { const lv = levelSelect.value; if (!lv) return alert("Select Level!"); currentLevel = lv; fileSelect.innerHTML = ''; try { const res = await fetch(getFileListUrl(lv)); const json = await res.json(); fileSelect.innerHTML = json.files.map(f => ``).join(''); } catch (e) { alert("Load list failed"); } }; document.getElementById('fileConfirm').onclick = async () => { const fn = fileSelect.value; if (!fn || fn.includes("Loading")) return; currentFileName = fn; const res = await fetch(getXlsxFileUrl(currentLevel, fn)); const buf = await res.arrayBuffer(); await parseExcel(buf, fn); }; document.getElementById('selectFileBtn').onclick = () => localFileInput.click(); localFileInput.onchange = (e) => { const f = e.target.files[0]; document.getElementById('fileNameDisplay').textContent = f ? f.name : "No file selected"; }; document.getElementById('localFileConfirmBtn').onclick = async () => { const f = localFileInput.files[0]; if (!f) return alert("Select a file!"); currentFileName = f.name; const reader = new FileReader(); reader.onload = (e) => parseExcel(e.target.result, f.name); reader.readAsArrayBuffer(f); }; document.getElementById('filterBtn').onclick = () => { const day = parseInt(document.getElementById('dayNum').value); const type = document.getElementById('daySelect').value; if (type === 'all') filteredWords = [...allWords]; else filteredWords = allWords.filter(w => w.day === day); currentWordIdx = 0; showWord(); updateInfoTip(); }; document.getElementById('showAllBtn').onclick = () => { let text = allWords.map(w => `Day ${w.day}: ${w.word} - ${w.meaning}`).join('\n'); alert(text || "No data"); }; document.getElementById('daySelect').onchange = (e) => { const input = document.getElementById('dayNum'); if (e.target.value === 'all') { input.value = '--'; input.readOnly = true; } else { input.value = '1'; input.readOnly = false; } }; toggleMode("local"); });
